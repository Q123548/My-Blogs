# Vue 组件传值基础

# 1.父传子

// Father 组件
<template>
    <Son :Array="Array" />
</template>

<script> 
//引入子组件
import Son  from './Son'
export default{
data(){
    name:"Father",
    //注册子组件
    compoents:{Son},
    data(){
        return{
            Array:[
                {id:'001',name:"Wang"},
                {id:'002',name:"Wang"},
                {id:'003',name:"Wang"},
                {id:'004',name:"Wang"},
            ]
        }
    },
}
}

</script>  


// Son 组件
<template>
 <li v-for="arr in Array" :key="arr.id" > </li>
</template>

<script>  
export default{
 data(){
    name:"Son",
    //用props 来接受来自父亲的数据
    props:['Array'],
}
}
</script>

# 2.子传父

  1. 首先在父组件中定义方法
  2. 父组件再将方法传个子组件
  3. 子组件在需要传参的时候调用方法，父组件就能接收数据
// Father 组件
<template>
    <Son :Array="Array"  :add="add"/>
</template>

<script> 
//引入子组件
import Son  from './Son'
export default{
data(){
    name:"Father",
    //注册子组件
    compoents:{Son},
    data(){
        return{
            Array:[
                {id:'001',name:"Wang"},
                {id:'002',name:"Wang"},
                {id:'003',name:"Wang"},
                {id:'004',name:"Wang"},
            ]
        }
    },
    //定义方法
    methods: {
        add(x){
            console.log("我收到了数据",x)
        }
    }
}
}

</script>  


// Son 组件
<template>
    <input type="text" @keyup.enter="done">
 <li v-for="arr in Array" :key="arr.id" > </li>
 
</template>

<script>  
export default{
 data(){
    name:"Son",
    //用props 来接受来自父亲的数据
    props:['Array','add'],
    methods:{
        done(){
            const da = {id:'005',name:"sad"}
            this.add(da)
        }
    }
}
}

# 兄弟组件通信

全局事件总线（GlobalEventBus）：

一种组件间通信的方式，适用于任意组件间通信。全局事件总线是一种可以在任意组件间通信的方式，本质上就是一个对象。它必须满足以下条件：

  1. 所有的组件对象都必须能看见他
  2. 这个对象必须能够使用$on、$emit和$off方法去绑定、触发和解绑事件

在mian.js中安装全局事件总线：

	beforeCreate() {
   		Vue.prototype.$bus = this //安装全局事件总线，$bus就是当前应用的vm
   	},

使用事件总线

接收数据：A组件想接收数据，则在A组件中给$bus绑定自定义事件，事件的回调留在A组件自身

    methods(){
        demo(data){...}
    },
    mounted() {
        this.$bus.$on('xxx',this.demo)
    }
// 提供数据：this.$bus.$emit('xxx',data)

最好在beforeDestroy钩子中，用$off去解绑当前组件所用到的事件
